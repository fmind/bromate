71def init_agent_from_config(config: AgentConfig) -> Agent:
72    """Initialize a model from config."""
73    api_key = config.api_key.get_secret_value() if config.api_key else None
74    genai.configure(api_key=api_key)  # global assignment!
75    gen_config = GenerationConfig(
76        temperature=config.temperature,
77        candidate_count=config.candidate_count,
78        max_output_tokens=config.max_output_tokens,
79    )
80    model = Agent(
81        model_name=config.name,
82        generation_config=gen_config,
83        system_instruction=config.system_instructions,
84    )
85    return model
