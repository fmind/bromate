40def init_driver_from_config(config: DriverConfig) -> Driver:
41    """Initialize the driver from config."""
42    driver: Driver  # not assiged!
43    if config.name == "Chrome":
44        driver = wd.Chrome(
45            options=wd.ChromeOptions(),
46            service=wd.ChromeService(),
47            keep_alive=config.keep_alive,
48        )
49    elif config.name == "Firefox":
50        driver = wd.Firefox(
51            options=wd.FirefoxOptions(),
52            service=wd.FirefoxService(),
53            keep_alive=config.keep_alive,
54        )
55    else:
56        raise ValueError(
57            f"Cannot initialize driver from config (unknown driver name): {config.name}!"
58        )
59    if config.maximize_window is True:
60        driver.maximize_window()
61    return driver

 28    def __init__(self, webelement: WebElement) -> None:
 29        """Constructor. A check is made that the given element is, indeed, a
 30        SELECT tag. If it is not, then an UnexpectedTagNameException is thrown.
 31
 32        :Args:
 33         - webelement - SELECT element to wrap
 34
 35        Example:
 36            from selenium.webdriver.support.ui import Select \n
 37            Select(driver.find_element(By.TAG_NAME, "select")).select_by_index(2)
 38        """
 39        if webelement.tag_name.lower() != "select":
 40            raise UnexpectedTagNameException(f"Select only works on <select> elements, not on {webelement.tag_name}")
 41        self._el = webelement
 42        multi = self._el.get_dom_attribute("multiple")
 43        self.is_multiple = multi and multi != "false"
 44
 45    @property
 46    def options(self) -> List[WebElement]:
 47        """Returns a list of all options belonging to this select tag."""
 48        return self._el.find_elements(By.TAG_NAME, "option")

 50    @property
 51    def all_selected_options(self) -> List[WebElement]:
 52        """Returns a list of all selected options belonging to this select
 53        tag."""
 54        return [opt for opt in self.options if opt.is_selected()]
 55
 56    @property
 57    def first_selected_option(self) -> WebElement:
 58        """The first selected option in this select tag (or the currently
 59        selected option in a normal select)"""

 62                return opt

 65    def select_by_value(self, value: str) -> None:
 66        """Select all options that have a value matching the argument. That is,
 67        when given "foo" this would select an option like:
 68
 69        <option value="foo">Bar</option>
 70
 71        :Args:
 72         - value - The value to match against
 73
 74        throws NoSuchElementException If there is no option with specified value in SELECT
 75        """
 76        css = f"option[value ={self._escape_string(value)}]"
 77        opts = self._el.find_elements(By.CSS_SELECTOR, css)
 78        matched = False
 79        for opt in opts:
 80            self._set_selected(opt)
 81            if not self.is_multiple:
 82                return
 83            matched = True
 84        if not matched:
 85            raise NoSuchElementException(f"Cannot locate option with value: {value}")
 86
 87    def select_by_index(self, index: int) -> None:
 88        """Select the option at the given index. This is done by examining the
 89        "index" attribute of an element, and not merely by counting.
 90
 91        :Args:
 92         - index - The option at this index will be selected
 93
 94        throws NoSuchElementException If there is no option with specified index in SELECT
 95        """
 96        match = str(index)
 97        for opt in self.options:
 98            if opt.get_attribute("index") == match:
 99                self._set_selected(opt)
100                return
101        raise NoSuchElementException(f"Could not locate element with index {index}")
102
103    def select_by_visible_text(self, text: str) -> None:
104        """Select all options that display text matching the argument. That is,
105        when given "Bar" this would select an option like:
106
107         <option value="foo">Bar</option>
108
109        :Args:
110         - text - The visible text to match against
111
112         throws NoSuchElementException If there is no option with specified text in SELECT
113        """
114        xpath = f".//option[normalize-space(.) = {self._escape_string(text)}]"
115        opts = self._el.find_elements(By.XPATH, xpath)
116        matched = False
117        for opt in opts:
118            self._set_selected(opt)
119            if not self.is_multiple:
120                return
121            matched = True
122
123        if len(opts) == 0 and " " in text:
124            sub_string_without_space = self._get_longest_token(text)
125            if sub_string_without_space == "":
126                candidates = self.options
127            else:
128                xpath = f".//option[contains(.,{self._escape_string(sub_string_without_space)})]"
129                candidates = self._el.find_elements(By.XPATH, xpath)
130            for candidate in candidates:
131                if text == candidate.text:
132                    self._set_selected(candidate)
133                    if not self.is_multiple:
134                        return
135                    matched = True
136
137        if not matched:
138            raise NoSuchElementException(f"Could not locate element with visible text: {text}")

211    def _set_selected(self, option) -> None:
212        if not option.is_selected():
213            if not option.is_enabled():
214                raise NotImplementedError("You may not select a disabled option")
215            option.click()
216
217    def _unset_selected(self, option) -> None:
218        if option.is_selected():
219            option.click()

221    def _escape_string(self, value: str) -> str:
222        if '"' in value and "'" in value:
223            substrings = value.split('"')
224            result = ["concat("]
225            for substring in substrings:
226                result.append(f'"{substring}"')
227                result.append(", '\"', ")
228            result = result[0:-1]
229            if value.endswith('"'):
230                result.append(", '\"'")
231            return "".join(result) + ")"
232
233        if '"' in value:
234            return f"'{value}'"
235
236        return f'"{value}"'
237
238    def _get_longest_token(self, value: str) -> str:
239        items = value.split(" ")
240        longest = ""
241        for item in items:
242            if len(item) > len(longest):
243                longest = item
244        return longest
